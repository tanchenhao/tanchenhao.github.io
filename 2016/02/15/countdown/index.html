canvas绘图

canvas是一种基于状态的绘图，先确定状态，再进行绘制。

需要先获得context进行绘制
var canvas = document.getElementById(‘canvans’);
var context=canvans.getContext(‘2d’);

画直线stroke笔画
context.moveTo(100,100)//起点左上角为原点
context.lineTo(700,700)//终点
context.lineWidth = 5 //线宽
context.strokeStyle = ‘#005588’//线的样式
context.stroke()//画线

填充
context.fillStyle = (‘’)
context.fill()

定义图形分割，使几个图形状态互相不干扰
context.beginPath() //重新规划路径
context.closePath() //结束路径，会将不封闭路径连接起来

js时间方法

new Date(2016, 1, 15, 18, 47, 52)//定义一个时间量，其中月份从0开始计。
new Date()//获得当前时间。
getTime()//方法获得距离1970年1月1日之间的毫秒数。

随机量的取得

Math.random()随机取0到1之间的数值。
Math.pow(-1, Math.ceil(Math.random()*1000)) 随机取-1或者1。
ceil()向上取最接近的整数
floor()向下取最接近的整数
pow（-1，9）表示-1的9次幂。

源码分析：

var WINDOW_WIDTH = 1024;
var WINDOW_HEIGHT = 768;
var RADIUS = 8;  //定义小圆球半径
var MARGIN_TOP = 60; 
var MARGIN_LEFT = 30;

const endTime = new Date(2016, 1, 15, 18, 47, 52);//设定倒计时的目标时间
var curShowTimeSeconds = 0; //定义需要倒计时总秒数curShowTimeSeconds 
var balls = [];//定义数字改变时掉落球的数组
const colors = ["#33B5E5", "#0099CC", "#AA66CC", "#9933CC", "#99CC00", "#669900", "#FFBB33", "#FF8800", "#FF4444", "#CC0000"];
//定义存放掉落小球颜色的数组

window.onload = function(){
    var canvas = document.getElementById('canvas');
    var context = canvas.getContext("2d"); 
    //设定画布长和高
    canvas.width = WINDOW_WIDTH;    
    canvas.height = WINDOW_HEIGHT;

    curShowTimeSeconds = getCurrentShowSeconds();//调用函数

    //动画绘制方法
    setInterval(
        function(){
        //每一帧要做的事情
            render(context); //绘制当前画面
            update(); //根据当前画面，调整数据结构
        }
        ,
        50//时间，单位毫秒
        );

}

function getCurrentShowSeconds () {
    //获取倒计时显示的秒数
    var curTime = new Date();//定义当前时间
    var ret = endTime.getTime() - curTime.getTime();//getTime()方法获得距离1970年的毫秒数
    ret = Math.round(ret/1000);//将毫秒转换为秒
    return ret >= 0 ? ret : 0;//如果ret 大于0则返回ret，否则返回0.
}

function update(){
    //动画的具体行为
    var nextShowTimeSeconds = getCurrentShowSeconds();//获得倒计时此时的秒数

    //将现在倒计时上，小时，分钟，秒显示的具体数字
    var nextHours = parseInt(nextShowTimeSeconds/3600);
    var nextMinutes = parseInt((nextShowTimeSeconds-3600*nextHours)/60);
    var nextSeconds = nextShowTimeSeconds%60;

    var curHours = parseInt(curShowTimeSeconds/3600);
    var curMinutes = parseInt((curShowTimeSeconds-3600*curHours)/60);
    var curSeconds = curShowTimeSeconds%60;

    if(nextSeconds != curSeconds){
        //判断时间是否更改，若改变为新的秒数
        curShowTimeSeconds = nextShowTimeSeconds;

        //获得掉落小球的形成位置
        if(parseInt(curHours/10) != parseInt(nextHours/10)){
            addBalls(MARGIN_LEFT, MARGIN_TOP, parseInt(curHours/10));
        }
        if(parseInt(curHours%10) != parseInt(nextHours%10)){
            addBalls(MARGIN_LEFT+15*(RADIUS+1), MARGIN_TOP, parseInt(curHours%10));
        }
        if(parseInt(curMinutes/10) != parseInt(nextMinutes/10)){
            addBalls(MARGIN_LEFT+39*(RADIUS+1), MARGIN_TOP, parseInt(curMinutes/10));
        }
        if(parseInt(curMinutes%10) != parseInt(nextMinutes%10)){
            addBalls(MARGIN_LEFT+54*(RADIUS+1), MARGIN_TOP, parseInt(curMinutes%10));
        }
        if(parseInt(curSeconds/10) != parseInt(nextSeconds/10)){
            addBalls(MARGIN_LEFT+78*(RADIUS+1), MARGIN_TOP, parseInt(curSeconds/10));
        }
        if(parseInt(curSeconds%10) != parseInt(nextSeconds%10)){
            addBalls(MARGIN_LEFT+93*(RADIUS+1), MARGIN_TOP, parseInt(curSeconds%10));
        }

    }
    updateBalls();//启动小球的掉落
}

function updateBalls(){
    //小球掉落的动画
    for(var i=0; i<balls.length; i++){

        balls[i].x += balls[i].vx; //小球的x轴位置=小球沿x轴速度的累加
        balls[i].y += balls[i].vy; //小球的y轴位置=小球沿x轴速度的累加
        balls[i].vy += balls[i].g; //小球的y轴的速度=小球加速度的累加

        if(balls[i].y >= WINDOW_HEIGHT-RADIUS){
            //碰撞检测
            balls[i].y = WINDOW_HEIGHT-RADIUS;//小球落地后不足下降
            balls[i].vy = -balls[i].vy*0.75;//小球弹球的速度=小球原来的速度反向*空气摩擦系数
        }
    }
}

function addBalls (x, y, num) {
    //在balls数组中生成掉落的小球
    for( var i = 0; i<digit[num].length; i++){ //遍历阵列的行
        for(var j = 0; j<digit[num][i].length; j++){//遍历阵列的行中的每一个数
            if( digit[num][i][j] == 1){ //判断值是否为1
                var aBall = {
                    //定义对象掉落的小球
                        x:x+j*2*(RADIUS+1)+(RADIUS+1), 
                        y:y+i*2*(RADIUS+1)+(RADIUS+1), 
                        g:1.5+Math.random(),//在1.5到2.5之间随机取g得值
                        vx:Math.pow(-1, Math.ceil(Math.random()*1000))*4,//随机取x轴速度的方向
                        vy:-5,
                        color:colors[Math.floor(Math.random()*colors.length)]//随机定义小球的颜色
                        }
                balls.push( aBall )//往balls数组中，放入小球
            }

        }
    }
}

function render(cxt){

    cxt.clearRect(0, 0, WINDOW_WIDTH, WINDOW_HEIGHT);//清空屏幕

    //定义倒计时各个位置上的数值
    var hours = parseInt(curShowTimeSeconds/3600);
    var minutes = parseInt((curShowTimeSeconds-3600*hours)/60);
    var seconds = parseInt(curShowTimeSeconds%60);
    //以小球的形式绘制数字
    renderDight(MARGIN_LEFT, MARGIN_TOP, parseInt(hours/10), cxt);
    renderDight(MARGIN_LEFT+15*(RADIUS+1), MARGIN_TOP, parseInt(hours%10), cxt);
    renderDight(MARGIN_LEFT+30*(RADIUS+1), MARGIN_TOP, 10, cxt);
    renderDight(MARGIN_LEFT+39*(RADIUS+1), MARGIN_TOP, parseInt(minutes/10), cxt);
    renderDight(MARGIN_LEFT+54*(RADIUS+1), MARGIN_TOP, parseInt(minutes%10), cxt);
    renderDight(MARGIN_LEFT+69*(RADIUS+1), MARGIN_TOP, 10, cxt);
    renderDight(MARGIN_LEFT+78*(RADIUS+1), MARGIN_TOP, parseInt(seconds/10), cxt);
    renderDight(MARGIN_LEFT+93*(RADIUS+1), MARGIN_TOP, parseInt(seconds%10), cxt);

    for(var i=0; i<balls.length; i++){
        //绘制掉落的小球
        cxt.fillStyle=balls[i].color;//随机定义掉落小球的颜色

        cxt.beginPath();
        cxt.arc( balls[i].x, balls[i].y, RADIUS, 0, 2*Math.PI);
        cxt.closePath();
        cxt.fill();
    }
}
function renderDight(x, y, num, cxt){
    //绘制显示倒计时的小球
    cxt.fillStyle = "rgb(0, 102, 155)";

    for( var i = 0; i<digit[num].length; i++){
        for(var j = 0; j<digit[num][i].length; j++){
            if( digit[num][i][j] == 1){
                cxt.beginPath();
                cxt.arc( x+j*2*(RADIUS+1)+(RADIUS+1), y+i*2*(RADIUS+1)+(RADIUS+1), RADIUS, 0, 2*Math.PI);
                cxt.closePath();

                cxt.fill();
            }
        }
    }
}
